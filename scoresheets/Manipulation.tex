
The maximum time for this test is 3 minutes.

\begin{scorelist}
	\scoreheading{Grasping objects}
	\scoreitem[5]{10}{For each successful grasp of any object (lifting it up to at least 5 cm for more than 10 seconds)}

	\scoreheading{Placing objects}
	\scoreitem[5]{10}{For each successful placement of any object (safely stands still for more than 10 seconds)}

	\scoreheading{Recognizing objects}
	\scoreitem[5]{10}{Every correctly recognized object in the report file}
	\scoreitem[5]{-5}{False positive label}

	\scoreheading{Hidden object optional}
	\scoreitem[1]{20}{Finding a hidden or occluded object}
	\scoreitem[1]{20}{Successful grasp of hidden or occluded object}

	\scoreheading{Opening door or drawer}
	\scoreitem[1]{50}{Open the drawer or door}
	
	\setTotalScore{240}
\end{scorelist}
