
The maximum time for this test is 3 minutes.

\begin{scorelist}
	\scoreitem{5}{Understand which table to clean}
	\scoreitem{5}{Navigate to the correct table}
	
	\scoreitem[3]{10}{For each successful grasp of any object (lifting it up to at least 5 cm for more than 10 seconds)}
	
	\scoreitem[3]{10}{For each successful placement of any object (safely stands still for more than 10 seconds)}
	
	\scoreitem{30}{Opening the closet door}
	
	% Start group of mutually exclusive scores
	\scoreitem[3]{10}{Correctly recognizing an object}
	\scoreitem[3]{-5}{Incorrectly recognizing an object}
	% end group of mutually exclusive scores
	
	% Start group of mutually exclusive scores
	\scoreitem[3]{10}{Correctly state an object is unknown}
	\scoreitem[3]{-5}{Incorrectly state an object is unknown}
	% end group of mutually exclusive scores
	
	\scoreitem[3]{10}{Correctly describe an unknown object}
	
	\scoreitem[3]{10}{Storing an object in the closet} % Bonus after achieving a complete sequence of all subtasks
	\scoreitem{5}{Robot brings cleaning tool itself}
	
	\scoreitem{30}{Cleaning the table of food clutter} % TODO: Maybe score this relative to amount of clutter removed. 
	
	\setTotalScore{250}
\end{scorelist}
